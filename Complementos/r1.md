1. O que será escrito pelo programa C abaixo quando for executado?

```c
#include <stdio.h>

int p1(int n, int v[n], int c) {
  int a = -1;
  for (int i = 0; i < n; i++) {
    if (v[i] > c) {
      a = i;
      break;
    }
  }
  return a;
}

int p2(int n, int v[n], int c) {
  int a = -1;
  for (int i = 0; i < n; i++) {
    if (v[i] < c) {
      a = i;
    }
  }
  return a;
}

void p3(int n, int v[n]) {
  for (int i = 1; i < n-1; i++) {
    printf("%d ", v[i]);
  }
  putchar('\n');
}

void p4(int n, int v[n], int c) {
  int a, b;
  a = p1(n, v, c);
  b = p2(n, v, c);
  if (a < b && a != -1) {
    int d = v[a];
    v[a] = v[b];
    v[b] = d;
  }
}

int main() {
  int m[] = { 0, 8, 18, 4, 3, 14, 13, 10 };
  p3(8, m);
  p4(8, m, m[7]);
  p3(8, m);
  p4(8, m, m[7]);
  p3(8, m);
}
```
